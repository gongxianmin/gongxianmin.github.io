beforeRouteEnter钩子函数
``` javascript
beforeRouteEnter (to, from, next) {
this.$http.get(url).then((res) => {
// 此时vue的实例没有生成，如果要进行数据的绑定，必须在next函数中执行
next(vm => {
res = res.data
if (res && res.code === ‘200) {
vm.list = res.data
}
})
})
}
